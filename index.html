const express = require("express"); const app = express(); app.use(express.json()); let siparisler = []; // ANA SAYFA (MÜŞTERİ) app.get("/", (req, res) => { res.send(`

MotoPro Kurye



Giriş `); }); // LOGIN app.get("/login", (req, res) => { res.send(`

Giriş



`); }); // ADMIN app.get("/admin", (req, res) => { res.send(`

Admin Panel

`); }); // KURYE app.get("/kurye", (req, res) => { res.send(`

Kurye Panel

`); }); // API app.post("/siparis", (req,res)=>{ let s = req.body; s.id = Date.now(); s.durum = "Bekliyor"; siparisler.push(s); res.json(s); }); app.get("/siparisler",(req,res)=>{ res.json(siparisler); }); app.post("/kabul/:id",(req,res)=>{ let s = siparisler.find(x=>x.id==req.params.id); if(s) s.durum="Yolda"; res.json(s); }); app.post("/teslim/:id",(req,res)=>{ let s = siparisler.find(x=>x.id==req.params.id); if(s) s.durum="Teslim Edildi"; res.json(s); }); app.post("/sil/:id",(req,res)=>{ siparisler = siparisler.filter(x=>x.id!=req.params.id); res.json({ok:true}); }); app.listen(3000,()=>console.log("Çalışıyor: http://localhost:3000"));
